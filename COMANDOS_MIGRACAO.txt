# COMANDOS MIGRAÇÃO - AMBIENTE DESENVOLVIMENTO

# 1. Conectar ao Supabase
supabase link --project-ref <seu-project-ref>

# 2. Executar migração (functions, triggers, policies)
supabase db push

# 3. Verificar diferenças
supabase db diff

# 4. (Opcional) Deploy Edge Functions
supabase functions deploy asaas-webhook-charges
supabase functions deploy evolution-proxy
supabase functions deploy send-invite-email
# ... outras functions conforme necessário
